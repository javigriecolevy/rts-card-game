enum class AttributeCategory {
    COST,
    CLASS,
    TRIBE,
    BATTLECRY
    // Add more categories here
};

// Helper function to automatically assign unique bit positions
template <AttributeCategory category>
struct AttributeFlag {
    static constexpr int get() {
        return 1 << (static_cast<int>(category) * 8 + currentOffset++);
    }

    // A static variable to keep track of the offset within each category
    static inline int currentOffset = 0;
};

// Automatically generate attributes for each category
enum class CostAttribute {
    COST_1 = AttributeFlag<AttributeCategory::COST>::get(),
    COST_2 = AttributeFlag<AttributeCategory::COST>::get(),
    COST_3 = AttributeFlag<AttributeCategory::COST>::get(),
    COST_4 = AttributeFlag<AttributeCategory::COST>::get()
};

enum class ClassAttribute {
    MAGE = AttributeFlag<AttributeCategory::CLASS>::get(),
    WARRIOR = AttributeFlag<AttributeCategory::CLASS>::get()
};

enum class TribeAttribute {
    BEAST = AttributeFlag<AttributeCategory::TRIBE>::get(),
    DRAGON = AttributeFlag<AttributeCategory::TRIBE>::get()
};

enum class BattlecryAttribute {
    DESTROY_PIRATE = AttributeFlag<AttributeCategory::BATTLECRY>::get()
};
